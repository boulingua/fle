{{- /* vgwort/url.html — single source of truth for a page's VG Wort Zählmarke. Returns the fully-resolved counting-pixel URL for THIS page, or "" if the page carries no registered Zählmarke. Consumed by BOTH the preload (head/extensions.html) and the body (vgwort partial), so the two can never disagree. Token lookup order: 1. per-page front matter: `vgwort_pixel:` / `vgwort:` 2. data/vgwort.yaml entry matched by `url:` (== RelPermalink, base stripped) or by `path:` (== content/); token read from `pixel_url`, `public_id` or `token`. Token forms accepted: full https URL, `vgNN.met.vgwort.de/na/`, or a bare id (defaults to https://vg09.met.vgwort.de/na/). Paginated list continuations (/page/2/ …) return "" so each Zählmarke maps to exactly one URL. */ -}} {{- $url := "" -}} {{- $skip := false -}} {{- if in (slice "section" "taxonomy" "term") .Kind -}} {{- with .Paginator -}}{{- if gt .PageNumber 1 -}}{{- $skip = true -}}{{- end -}}{{- end -}} {{- end -}} {{- if not $skip -}} {{- $token := or .Params.vgwort_pixel .Params.vgwort -}} {{- if not $token -}} {{- $rel := .RelPermalink -}} {{- $base := site.BaseURL | replaceRE "https?://[^/]+" "" | strings.TrimRight "/" -}} {{- if $base -}}{{- $rel = $rel | strings.TrimPrefix $base -}}{{- end -}} {{- $fp := "" -}}{{- with .File -}}{{- $fp = printf "content/%s" .Path -}}{{- end -}} {{- $entries := index site.Data "vgwort" -}} {{- if reflect.IsSlice $entries -}} {{- range $entries -}} {{- if or (and .url (eq .url $rel)) (and .path (eq .path $fp)) -}} {{- $token = or .pixel_url .public_id .token -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- with $token -}} {{- $t := printf "%s" . -}} {{- if or (hasPrefix $t "http://") (hasPrefix $t "https://") -}}{{- $url = $t -}} {{- else if findRE "^vg[a-z0-9]+\\.met\\.vgwort\\.de/" $t -}}{{- $url = printf "https://%s" $t -}} {{- else -}}{{- $url = printf "https://vg09.met.vgwort.de/na/%s" $t -}} {{- end -}} {{- end -}} {{- end -}} {{- return $url -}}
